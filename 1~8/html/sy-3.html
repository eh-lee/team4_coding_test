문제 설명 단어 s의 가운데 글자를 반환하는 함수, solution을 만들어 보세요. 단어의 길이가 짝수라면 가운데 두글자를 반환하면 됩니다. 재한사항 s는 길이가 1 이상, 100이하인 스트링입니다. 입출력 예 s return "abcde" "c" "qwer" "we" function solution(s) { if (s.length % 2 != 0){ // 배열 길이 / 2 가 0이 아닐 때! let b = s[Math.floor(s.length / 2)]; //길이 5/2= 2.5 버림으로 2 (c) var answer = b; } else { let b = s[Math.floor(s.length / 2 - 1)] + s[Math.floor(s.length / 2)]; //길이 4/2=2 (e) 2-1=1 (w) => (w)+(e) var answer = b; } return answer; } Math.floor() : 소수점 이하를 버림한다. console.log(Math.floor(5.95)); // Expected output: 5 console.log(Math.floor(-5.05)); // Expected output: -6 Math.ceil() : 소수점 이하를 올림한다. Math.round() : 소수점 이하를 반올림한다.
